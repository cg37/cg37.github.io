public:
ListNode* removeElement(ListNode* head, int target) {
ListNode dummyNode = new ListNode(0);
dummyNode->next = head;
ListNode* cur = dummyNode;
while(cur->next!=NULL) {
if(cur->next->val == target) {
ListNode tmp = cur->next;
cur->next = cur->next->next;
delete tmp;
} else {
cur = cur ->next;
}
}

delete dummyHead;
return head;
}
